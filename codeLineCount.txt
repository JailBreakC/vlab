$(function(){
    //发送数据
    if(window.transfer)
        throw "变量名冲突：请检查是transfer变量是否被占用";
    
    window.transfer = function(){
        return {
            sendText : function(save, id, data, fn, norefresh) {
                    $.ajax({
                        type:"post",
                        url:"adminDataAPI.php",
                        data:{'save':save, 'id':id, 'data':data},
                        dataType:"text",
                        success:function(ret){
                            if(ret === 'success'){
                                fn&&fn(data);
                                alert("更新成功");
                                if(!norefresh)
                                    window.location.reload()
                            }else{
                                alert("更新失败");
                            }
                        },
                        error:function(ret){
                            alert("网络故障，稍后重试 " + ret);
                        }
                    });
                },
            //获取数据
            getText : function(query, id, fn, field) {
                field = field || '';
                var data;
                console.log(field);
                $.ajax({
                        type:"post",
                        url:"adminDataAPI.php",
                        data:{'query':query, 'id':id, 'field':field},
                        dataType:"text",
                        success:function(ret){
                            //console.log(ret);
                            data = JSON.parse(ret);
                            fn(data);
                        },
                        error:function(ret){
                            alert("网络故障，稍后重试 " + ret);
                        }
                });
                return data;
            },
            //删除数据
            deleteText : function(deleteTable, id, norefresh) {
                $.ajax({
                        type:"post",
                        url:"adminDataAPI.php",
                        data:{'delete':deleteTable, 'id':id},
                        dataType:"text",
                        success:function(ret){
                            if(ret === 'success'){
                                if(!norefresh){
                                    alert("删除成功");
                                    window.location.reload()
                                }
                            }else{
                                alert("删除失败");
                            }
                        },
                        error:function(ret){
                            alert("网络故障，稍后重试 " + ret);
                        }
                });
            }
        }
    }
});

                湖南科技大学矿业工程虚拟仿真实验教学中心源于原煤炭工业部在江南开设的唯一本科院校湘潭矿业学院计算机中心，该中心成立于1988年，是最早进行虚拟仿真实验教学的单位之一。中心在当时中心主任李仁发教授带领下率先开展虚拟仿真实验教学的应用和研究，相继承担国家自然科学基金项目“基于虚拟原型的虚拟实验室实现原理研究”(批准号:66974031)和教育部骨干教师基金项目“虚拟实验室技术”。2003年湘潭工学院计算机中心与湘潭师范学院计算机中心合并组建湖南科技大学计算机中心，为全校3万师生提供计算机教学公共教学服务平台，并持续推广虚拟仿真实验教学和开展虚拟仿真实验教学的应用研究。
